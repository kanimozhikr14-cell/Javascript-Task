<!-- Part 3 – Conditional Statements

If marks >= 90 → Grade A

If marks >= 50 → Grade C
Else → Fail
Use if else if. -->

    let mark = prompt("Enter the scored mark")
    if(mark >= 90){

        console.log(`${mark} is Grade A`);
        
    }
    else if(mark >= 75) {
        
        console.log(`${mark} is Grade B`);

    }
    else if(mark >= 50) {
        
        console.log(`${mark} is Grade C`);

    }
    else{
        console.log(`${mark} is Fail`);
        
    }
</script>

<!-- Q9: Traffic Signal Program using switch case:
-->

<script>
    let signal=prompt("Enter the color Red or Yellow or Green(First letter in Caps)")
    switch(signal){
        case "Red":console.log("STOP"); break;
        case "Yellow":console.log("Ready"); break;
        case "Green":console.log("GOO"); break;
        default: console.log("Invalid signal");break;
        
    }
</script>
